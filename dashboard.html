// dashboard.js - Versão Segura import { auth, db } from './firebase-config.js'; import { signOut, onAuthStateChanged } from 'https://www.gstatic.com/firebasejs/10.7.1/firebase-auth.js'; import { doc, getDoc, setDoc, updateDoc, increment, collection, addDoc, query, where, getDocs, orderBy, serverTimestamp } from 'https://www.gstatic.com/firebasejs/10.7.1/firebase-firestore.js'; let currentUser = null; // Verifica autenticação onAuthStateChanged(auth, async (user) => { if (!user) { // Se não tiver usuário, manda pro login window.location.href = 'index.html'; return; } // Se chegou aqui, está logado! currentUser = user; // Mostra o e-mail na tela para confirmar if(document.getElementById('userEmail')) { document.getElementById('userEmail').textContent = user.email.split('@')[0]; } loadBalance(); loadHistory(); }); // Carregar Saldo async function loadBalance() { try { const docRef = doc(db, "users", currentUser.uid); const docSnap = await getDoc(docRef); if (docSnap.exists()) { document.getElementById('coins').textContent = docSnap.data().coins || 0; } else { // Cria perfil inicial se não existir await setDoc(docRef, { coins: 0 }); document.getElementById('coins').textContent = 0; } } catch (e) { console.error("Erro ao carregar saldo:", e); } } // === LÓGICA DO ANÚNCIO E SALVAR TREINO === window.startAdProcess = function() { const modal = document.getElementById('adModal'); const timer = document.getElementById('timer'); let timeLeft = 5; modal.style.display = 'flex'; timer.textContent = timeLeft; const interval = setInterval(() => { timeLeft--; timer.textContent = timeLeft; if (timeLeft <= 0) { clearInterval(interval); modal.style.display = 'none'; saveWorkout(); // Salva só depois do tempo } }, 1000); } async function saveWorkout() { const type = document.getElementById('workoutType').value; const value = parseInt(document.getElementById('intensity').value); try { await addDoc(collection(db, "history"), { uid: currentUser.uid, type: type, value: value, date: serverTimestamp() }); await updateDoc(doc(db, "users", currentUser.uid), { coins: increment(value) }); alert(`Sucesso! Você ganhou ${value} moedas.`); loadBalance(); // Atualiza saldo visual loadHistory(); // Atualiza lista } catch (e) { alert("Erro ao salvar: " + e.message); } } async function loadHistory() { const list = document.getElementById('historyList'); if(!list) return; const q = query(collection(db, "history"), where("uid", "==", currentUser.uid), orderBy("date", "desc")); const querySnapshot = await getDocs(q); list.innerHTML = ""; querySnapshot.forEach((doc) => { const data = doc.data(); list.innerHTML += `
${data.type.toUpperCase()} +${data.value}
`; }); } window.logout = () => { signOut(auth).then(() => { window.location.href = 'index.html'; }); };
